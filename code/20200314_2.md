# 建立 array
```
import numpy as np
np.linspace(2.0, 3.0, num=5, endpoint=False)
# array([2. , 2.2, 2.4, 2.6, 2.8])

import numpy as np
np.linspace(2.0, 3.0, num=5, retstep=True)
# (array([2.  , 2.25, 2.5 , 2.75, 3.  ]), 0.25)

import numpy as np
np.indices((3, 3))
# array([[[0, 0, 0],
#        [1, 1, 1],
#        [2, 2, 2]],
#
#       [[0, 1, 2],
#        [0, 1, 2],
#        [0, 1, 2]]])

import numpy as np
ar10=np.diag((2,1,4,6));
ar10
# array([[2, 0, 0, 0],
#        [0, 1, 0, 0],
#        [0, 0, 4, 0],
#        [0, 0, 0, 6]])

import numpy as np
a = np.array([0, 1, 2])
np.tile(a, 2)
# array([0, 1, 2, 0, 1, 2])

import numpy as np
a = np.array([0, 1, 2])
np.tile(a, (2, 2))
# array([[0, 1, 2, 0, 1, 2],
#        [0, 1, 2, 0, 1, 2]])

import numpy as np
a = np.array([0, 1, 2])
np.tile(a, (2, 1, 2))
# array([[[0, 1, 2, 0, 1, 2]],
#
#        [[0, 1, 2, 0, 1, 2]]])
```

# 直接生成使用genfromtxt（）方法建立堆棧
```
import csv
import numpy as np

x = '''1,3,2,3,1,2,3,4
2,4,5,0.6,5,6,7,8
3,7,8,9,9,10,11,12
4,1,1.1,1.2,13,14,15,16
'''
with open("abc.txt",mode="w",encoding="utf-8") as file:
  file.write(x)
file.close()

np.genfromtxt('abc.txt', delimiter=',', invalid_raise = False)
# array([[ 1. ,  3. ,  2. ,  3. ,  1. ,  2. ,  3. ,  4. ],
#        [ 2. ,  4. ,  5. ,  0.6,  5. ,  6. ,  7. ,  8. ],
#        [ 3. ,  7. ,  8. ,  9. ,  9. , 10. , 11. , 12. ],
#        [ 4. ,  1. ,  1.1,  1.2, 13. , 14. , 15. , 16. ]])
```
